console.log('login.html loaded, API_URL =', typeof API_URL, API_URL); function login() { console.log('login clicked'); const user = document.getElementById('u').value.trim(); const pass = document.getElementById('p').value.trim(); const cb = 'login_cb_' + Date.now(); window[cb] = function(res) { console.log('login response:', res); if (res && res.result === 'ok') { localStorage.setItem('role', res.role); localStorage.setItem('judge_id', res.judge_id || ''); location.href = 'index.html'; } else { alert('登入失敗'); } delete window[cb]; }; const s = document.createElement('script'); s.src = API_URL + '?action=login' + '&username=' + encodeURIComponent(user) + '&password=' + encodeURIComponent(pass) + '&callback=' + cb + '&_=' + Date.now(); document.body.appendChild(s); }
